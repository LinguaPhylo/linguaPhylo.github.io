data {
   D = readNexus ( file= "examples/hcv.nexus" );
   L = nchar ( sites= D );
   numGroups = 4 ;
   taxa = taxa ( taxa= D );
   w = ntaxa ( taxa )- 1 ;
}
model {
   π ~ Dirichlet ( conc= [ 3.0 , 3.0 , 3.0 , 3.0 ]);
   rates ~ Dirichlet ( conc= [ 1.0 , 2.0 , 1.0 , 1.0 , 2.0 , 1.0 ]);
   Q = gtr ( rates= rates , freq= π );
   shape ~ LogNormal ( meanlog= 0.0 , sdlog= 2.0 );
   siteRates ~ DiscretizeGamma ( ncat= 4 , reps= L , shape= shape );
   groupSizes ~ RandomComposition ( k= numGroups , n= w );
   firstValue ~ LogNormal ( meanlog= 9.0 , sdlog= 2.0 );
   Θ ~ ExpMarkovChain ( firstValue= firstValue , n= numGroups );
   ψ ~ SkylineCoalescent ( groupSizes= groupSizes , taxa= taxa , theta= Θ );
   D ~ PhyloCTMC ( Q= Q , mu= 7.9E-4 , siteRates= siteRates , tree= ψ );
}
